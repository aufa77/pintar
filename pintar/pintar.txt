    <script>
        // Data Management
        class InventoryManager {
            constructor() {
                this.items = this.loadItems();
                this.borrowData = this.loadBorrowData();
                this.init();
            }

            init() {
                this.setupEventListeners();
                this.renderAll();
            }

            loadItems() {
                const saved = localStorage.getItem('inventoryItems');
                if (saved) {
                    return JSON.parse(saved);
                }
                // Default items
                return [
                    { id: 1, name: 'Laptop', icon: 'fa-laptop', stock: 5, initialStock: 5 },
                    { id: 2, name: 'Proyektor', icon: 'fa-video', stock: 3, initialStock: 3 },
                    { id: 3, name: 'Kamera', icon: 'fa-camera', stock: 4, initialStock: 4 },
                    { id: 4, name: 'Speaker', icon: 'fa-volume-up', stock: 6, initialStock: 6 },
                    { id: 5, name: 'Mic', icon: 'fa-microphone', stock: 8, initialStock: 8 },
                    { id: 6, name: 'Tripod', icon: 'fa-tripod', stock: 5, initialStock: 5 },
                    { id: 7, name: 'Whiteboard', icon: 'fa-chalkboard', stock: 2, initialStock: 2 },
                    { id: 8, name: 'Printer', icon: 'fa-print', stock: 3, initialStock: 3 },
                    { id: 9, name: 'Scanner', icon: 'fa-scanner', stock: 2, initialStock: 2 },
                    { id: 10, name: 'Tablet', icon: 'fa-tablet-alt', stock: 7, initialStock: 7 },
                    { id: 11, name: 'Modem WiFi', icon: 'fa-wifi', stock: 10, initialStock: 10 },
                    { id: 12, name: 'Headset', icon: 'fa-headphones', stock: 12, initialStock: 12 }
                ];
            }

            loadBorrowData() {
                const saved = localStorage.getItem('borrowData');
                return saved ? JSON.parse(saved) : [];
            }

            saveItems() {
                localStorage.setItem('inventoryItems', JSON.stringify(this.items));
            }

            saveBorrowData() {
                localStorage.setItem('borrowData', JSON.stringify(this.borrowData));
            }

            setupEventListeners() {
                // Navigation
                document.querySelectorAll('.nav-item').forEach(item => {
                    item.addEventListener('click', (e) => {
                        const targetPage = e.currentTarget.dataset.page;
                        this.switchPage(targetPage);
                    });
                });

                // Buttons
                document.getElementById('borrow-btn').addEventListener('click', () => {
                    this.openBorrowModal();
                });

                document.getElementById('return-btn').addEventListener('click', () => {
                    this.openReturnModal();
                });

                document.getElementById('add-item-btn').addEventListener('click', () => {
                    this.openAddItemModal();
                });

                // Modals
                document.getElementById('close-borrow-modal').addEventListener('click', () => {
                    this.closeModal('borrow-modal');
                });

                document.getElementById('close-return-modal').addEventListener('click', () => {
                    this.closeModal('return-modal');
                });

                document.getElementById('close-add-item-modal').addEventListener('click', () => {
                    this.closeModal('add-item-modal');
                });

                // Forms
                document.getElementById('borrow-form').addEventListener('submit', (e) => {
                    e.preventDefault();
                    this.handleBorrow();
                });

                document.getElementById('add-item-form').addEventListener('submit', (e) => {
                    e.preventDefault();
                    this.handleAddItem();
                });

                // Close modal on outside click
                window.addEventListener('click', (e) => {
                    if (e.target.classList.contains('modal')) {
                        this.closeModal(e.target.id);
                    }
                });

                // Export button
                document.getElementById('export-btn').addEventListener('click', () => {
                    this.exportData();
                });
            }

            switchPage(pageName) {
                // Update nav
                document.querySelectorAll('.nav-item').forEach(item => {
                    item.classList.remove('active');
                });
                document.querySelector(`[data-page="${pageName}"]`).classList.add('active');

                // Update page
                document.querySelectorAll('.page').forEach(page => {
                    page.classList.remove('active');
                });
                document.getElementById(pageName).classList.add('active');

                // Render specific page content
                if (pageName === 'inventory') {
                    this.renderInventory();
                }
            }

            openBorrowModal() {
                this.renderBarangOptions();
                document.getElementById('borrow-modal').classList.add('active');
            }

            openReturnModal() {
                this.renderReturnList();
                document.getElementById('return-modal').classList.add('active');
            }

            openAddItemModal() {
                document.getElementById('add-item-modal').classList.add('active');
            }

            closeModal(modalId) {
                document.getElementById(modalId).classList.remove('active');
                if (modalId === 'borrow-modal') {
                    this.resetBorrowForm();
                }
            }

            renderBarangOptions() {
                const container = document.getElementById('barang-options');
                container.innerHTML = '';

                this.items.forEach(item => {
                    const card = document.createElement('div');
                    card.className = 'barang-card';
                    card.dataset.id = item.id;
                    
                    const stockClass = item.stock <= 2 ? 'low' : '';
                    const stockText = item.stock <= 2 ? `Hanya ${item.stock} tersisa!` : `Tersedia: ${item.stock}`;

                    card.innerHTML = `
                        <div class="barang-icon">
                            <i class="fas ${item.icon}"></i>
                        </div>
                        <div class="barang-name">${item.name}</div>
                        <div class="barang-stock ${stockClass}">${stockText}</div>
                    `;

                    if (item.stock === 0) {
                        card.style.opacity = '0.5';
                        card.style.cursor = 'not-allowed';
                    } else {
                        card.addEventListener('click', () => {
                            document.querySelectorAll('.barang-card').forEach(c => {
                                c.classList.remove('selected');
                            });
                            card.classList.add('selected');
                        });
                    }

                    container.appendChild(card);
                });
            }

            renderReturnList() {
                const container = document.getElementById('return-list');
                const emptyState = document.getElementById('no-borrowers');

                // Filter only active borrowings (not returned)
                const activeBorrowData = this.borrowData.filter(item => !item.status || item.status !== 'dikembalikan');

                if (activeBorrowData.length === 0) {
                    container.innerHTML = '';
                    emptyState.style.display = 'block';
                    return;
                }

                emptyState.style.display = 'none';
                container.innerHTML = '';

                // Group by borrower
                const borrowers = {};
                activeBorrowData.forEach(borrow => {
                    const key = `${borrow.nama}-${borrow.kelas}`;
                    if (!borrowers[key]) {
                        borrowers[key] = {
                            nama: borrow.nama,
                            kelas: borrow.kelas,
                            items: []
                        };
                    }
                    borrowers[key].items.push({
                        id: borrow.id,
                        name: borrow.barang,
                        quantity: borrow.quantity || 1,
                        itemId: borrow.barangId
                    });
                });

                // Render borrower cards
                Object.values(borrowers).forEach(borrower => {
                    const returnItem = document.createElement('div');
                    returnItem.className = 'return-item';

                    const initials = borrower.nama.split(' ').map(n => n[0]).join('').substring(0, 2).toUpperCase();

                    returnItem.innerHTML = `
                        <div class="return-info">
                            <div class="return-avatar">${initials}</div>
                            <div class="return-details">
                                <h4>${borrower.nama}</h4>
                                <p>Kelas: ${borrower.kelas}</p>
                            </div>
                        </div>
                        <div class="return-items">
                            ${borrower.items.map(item => 
                                `<span class="return-item-tag">${item.name} (${item.quantity})</span>`
                            ).join('')}
                        </div>
                    `;

                    returnItem.addEventListener('click', () => {
                        borrower.items.forEach(item => {
                            this.returnItem(item.id, item.itemId, item.quantity);
                        });
                        this.renderReturnList();
                        this.renderAll();
                        this.showToast('Semua barang berhasil dikembalikan!', 'success');
                    });

                    container.appendChild(returnItem);
                });
            }

            renderInventory() {
                const container = document.getElementById('inventory-grid');
                const emptyState = document.getElementById('inventory-empty');

                if (this.items.length === 0) {
                    container.innerHTML = '';
                    emptyState.style.display = 'block';
                    return;
                }

                emptyState.style.display = 'none';
                container.innerHTML = '';

                this.items.forEach(item => {
                    const card = document.createElement('div');
                    card.className = 'inventory-item';

                    const stockClass = item.stock <= 2 ? 'low' : '';
                    const stockBadge = item.stock === 0 ? 'badge-danger' : 
                                     item.stock <= 2 ? 'badge-warning' : 'badge-success';

                    card.innerHTML = `
                        <div class="inventory-icon">
                            <i class="fas ${item.icon}"></i>
                        </div>
                        <div class="inventory-name">${item.name}</div>
                        <div class="inventory-stock">Stok: <span class="badge ${stockBadge}">${item.stock}</span></div>
                        <div class="inventory-actions">
                            <button class="btn btn-xs btn-primary" onclick="inventoryManager.editItem(${item.id})">
                                <i class="fas fa-edit"></i>
                            </button>
                            <button class="btn btn-xs btn-danger" onclick="inventoryManager.deleteItem(${item.id})">
                                <i class="fas fa-trash"></i>
                            </button>
                        </div>
                    `;

                    container.appendChild(card);
                });
            }

            handleBorrow() {
                const selectedBarang = document.querySelector('.barang-card.selected');
                if (!selectedBarang) {
                    this.showToast('Pilih barang terlebih dahulu', 'error');
                    return;
                }

                const barangId = parseInt(selectedBarang.dataset.id);
                const barang = this.items.find(item => item.id === barangId);
                const nama = document.getElementById('nama-peminjam').value;
                const kelas = document.getElementById('kelas-peminjam').value;
                const jumlah = parseInt(document.getElementById('jumlah-pinjam').value);

                if (jumlah > barang.stock) {
                    this.showToast('Jumlah melebihi stok yang tersedia', 'error');
                    return;
                }

                const waktu = new Date().toLocaleString('id-ID');

                // Add to borrow data
                const newBorrow = {
                    id: Date.now(),
                    nama,
                    kelas,
                    barang: barang.name,
                    barangId,
                    quantity: jumlah,
                    waktu,
                    status: 'dipinjam'
                };

                this.borrowData.push(newBorrow);

                // Update stock
                barang.stock -= jumlah;

                // Save data
                this.saveBorrowData();
                this.saveItems();

                // Update UI
                this.renderAll();
                this.closeModal('borrow-modal');
                this.showToast('Peminjaman berhasil dicatat!', 'success');
            }

            handleAddItem() {
                const nama = document.getElementById('nama-barang').value;
                const stok = parseInt(document.getElementById('stok-barang').value);
                const icon = document.getElementById('icon-barang').value || 'fa-box';

                const newItem = {
                    id: Date.now(),
                    name: nama,
                    icon: icon,
                    stock: stok,
                    initialStock: stok
                };

                this.items.push(newItem);
                this.saveItems();
                this.renderAll();
                this.closeModal('add-item-modal');
                this.showToast('Barang berhasil ditambahkan!', 'success');

                // Reset form
                document.getElementById('add-item-form').reset();
            }

            editItem(id) {
                const item = this.items.find(i => i.id === id);
                if (!item) return;

                const newStock = prompt(`Edit stok untuk ${item.name}:`, item.stock);
                if (newStock !== null && !isNaN(newStock)) {
                    item.stock = parseInt(newStock);
                    this.saveItems();
                    this.renderAll();
                    this.showToast('Stok berhasil diperbarui!', 'success');
                }
            }

            deleteItem(id) {
                const item = this.items.find(i => i.id === id);
                if (!item) return;

                if (confirm(`Apakah Anda yakin ingin menghapus ${item.name}?`)) {
                    this.items = this.items.filter(i => i.id !== id);
                    this.saveItems();
                    this.renderAll();
                    this.showToast('Barang berhasil dihapus!', 'success');
                }
            }

            returnItem(borrowId, itemId, quantity) {
                // Find and update the borrow item status
                const borrowItem = this.borrowData.find(item => item.id === borrowId);
                if (borrowItem) {
                    borrowItem.status = 'dikembalikan';
                    borrowItem.waktuPengembalian = new Date().toLocaleString('id-ID');
                }

                // Update stock
                const item = this.items.find(i => i.id === itemId);
                if (item) {
                    item.stock += quantity;
                }

                // Save data
                this.saveBorrowData();
                this.saveItems();
            }

            renderPeminjamanTable() {
                const tbody = document.getElementById('peminjaman-table');
                const emptyState = document.getElementById('empty-state');

                if (this.borrowData.length === 0) {
                    tbody.innerHTML = '';
                    emptyState.style.display = 'block';
                    return;
                }

                emptyState.style.display = 'none';
                tbody.innerHTML = '';

                this.borrowData.forEach(borrow => {
                    const row = document.createElement('tr');
                    
                    // Determine status and badge
                    let statusBadge = '';
                    if (borrow.status === 'dikembalikan') {
                        statusBadge = '<span class="badge badge-success">SUDAH DIKEMBALIKAN</span>';
                    } else {
                        statusBadge = '<span class="badge badge-warning">DIPINJAM</span>';
                    }

                    row.innerHTML = `
                        <td>${borrow.nama}</td>
                        <td>${borrow.kelas}</td>
                        <td>${borrow.barang} (${borrow.quantity || 1})</td>
                        <td>${borrow.waktu}</td>
                        <td>${statusBadge}</td>
                        <td>
                            <div class="table-action">
                                ${borrow.status !== 'dikembalikan' ? 
                                    `<button class="btn btn-sm btn-success" onclick="inventoryManager.returnSingleItem(${borrow.id}, ${borrow.barangId}, ${borrow.quantity || 1})">
                                        <i class="fas fa-undo"></i> Kembalikan
                                    </button>` : ''}
                                <button class="btn btn-sm btn-danger" onclick="inventoryManager.deleteHistory(${borrow.id})">
                                    <i class="fas fa-trash"></i> Hapus
                                </button>
                            </div>
                        </td>
                    `;
                    tbody.appendChild(row);
                });
            }

            returnSingleItem(borrowId, itemId, quantity) {
                if (confirm('Apakah Anda yakin ingin mengembalikan barang ini?')) {
                    this.returnItem(borrowId, itemId, quantity);
                    this.renderAll();
                    this.showToast('Barang berhasil dikembalikan!', 'success');
                }
            }

            deleteHistory(id) {
                if (confirm('Apakah Anda yakin ingin menghapus riwayat ini?')) {
                    this.borrowData = this.borrowData.filter(item => item.id !== id);
                    this.saveBorrowData();
                    this.renderAll();
                    this.showToast('Riwayat berhasil dihapus!', 'success');
                }
            }

            updateDashboard() {
                // Calculate stats
                const totalBarang = this.items.reduce((sum, item) => sum + item.stock, 0);
                const totalInitial = this.items.reduce((sum, item) => sum + item.initialStock, 0);
                
                // Count only active borrowings (not returned)
                const activeBorrowData = this.borrowData.filter(item => !item.status || item.status !== 'dikembalikan');
                const barangDipinjam = activeBorrowData.reduce((sum, item) => sum + (item.quantity || 1), 0);
                const peminjamAktif = new Set(activeBorrowData.map(item => `${item.nama}-${item.kelas}`)).size;

                // Update DOM
                document.getElementById('total-barang').textContent = totalBarang;
                document.getElementById('barang-change').textContent = `${totalBarang}/${totalInitial}`;
                document.getElementById('barang-dipinjam').textContent = barangDipinjam;
                document.getElementById('peminjam-aktif').textContent = peminjamAktif;
                document.getElementById('jenis-barang').textContent = this.items.length;

                // Calculate today's transactions (both borrow and return)
                const today = new Date().toLocaleDateString('id-ID');
                const todayTransactions = this.borrowData.filter(item => 
                    item.waktu.includes(today) || (item.waktuPengembalian && item.waktuPengembalian.includes(today))
                ).length;
                document.getElementById('transaksi-hari').textContent = todayTransactions;

                // Find most popular item
                if (this.borrowData.length > 0) {
                    const itemCounts = {};
                    this.borrowData.forEach(item => {
                        itemCounts[item.barang] = (itemCounts[item.barang] || 0) + 1;
                    });
                    const mostPopular = Object.entries(itemCounts)
                        .sort((a, b) => b[1] - a[1])[0];
                    document.getElementById('barang-populer').textContent = mostPopular[0];
                } else {
                    document.getElementById('barang-populer').textContent = '-';
                }
            }

            renderAll() {
                this.updateDashboard();
                this.renderPeminjamanTable();
                this.renderInventory();
            }

            resetBorrowForm() {
                document.getElementById('borrow-form').reset();
                document.querySelectorAll('.barang-card').forEach(card => {
                    card.classList.remove('selected');
                });
            }

            showToast(message, type = 'success', title = null) {
                const toast = document.getElementById('toast');
                const toastIcon = document.getElementById('toast-icon');
                const toastTitle = document.getElementById('toast-title');
                const toastMessage = document.getElementById('toast-message');

                toastTitle.textContent = title || (type === 'success' ? 'Sukses' : type === 'error' ? 'Error' : 'Peringatan');
                toastMessage.textContent = message;

                if (type === 'success') {
                    toastIcon.className = 'toast-icon success';
                    toastIcon.innerHTML = '<i class="fas fa-check"></i>';
                } else if (type === 'error') {
                    toastIcon.className = 'toast-icon error';
                    toastIcon.innerHTML = '<i class="fas fa-times"></i>';
                } else {
                    toastIcon.className = 'toast-icon warning';
                    toastIcon.innerHTML = '<i class="fas fa-exclamation"></i>';
                }

                toast.classList.add('show');

                setTimeout(() => {
                    toast.classList.remove('show');
                }, 4000);
            }

            exportData() {
                const csvContent = [
                    ['Nama', 'Kelas', 'Barang', 'Jumlah', 'Waktu Peminjaman', 'Waktu Pengembalian', 'Status'],
                    ...this.borrowData.map(item => [
                        item.nama,
                        item.kelas,
                        item.barang,
                        item.quantity || 1,
                        item.waktu,
                        item.waktuPengembalian || '-',
                        item.status === 'dikembalikan' ? 'Sudah Dikembalikan' : 'Dipinjam'
                    ])
                ].map(row => row.join(',')).join('\n');

                const blob = new Blob([csvContent], { type: 'text/csv' });
                const url = window.URL.createObjectURL(blob);
                const a = document.createElement('a');
                a.href = url;
                a.download = `peminjaman_${new Date().toISOString().split('T')[0]}.csv`;
                a.click();
                window.URL.revokeObjectURL(url);

                this.showToast('Data berhasil diekspor!', 'success');
            }
        }

        // Initialize the application
        const inventoryManager = new InventoryManager();
    </script>
